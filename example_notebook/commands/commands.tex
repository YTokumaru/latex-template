    \verb|\R|   & \(\R\)   & 実数               \\
    \verb|\C|   & \(\C\)   & 複素数             \\
    \verb|\K|   & \(\K\)   & 上記の体のいずれか \\
    \verb|\Ima| & \(\Ima\) & 像                 \\
    \bottomrule
\end{longtable}
